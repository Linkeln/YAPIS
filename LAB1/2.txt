def calculate_total_tax(data_table)
    total = 0.0

    for i = 0; i < 5; i = i + 1
        current_row = data_table[i]
        

        price = (float) current_row["amount"]
        

        tax_amount = (price * 0.20) / 1.0
        total = total + tax_amount
    
    write("Total tax calculated")
    return total


sales_log = table("id", "amount")

final_tax = calculate_total_tax(sales_log)
